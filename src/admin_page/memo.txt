drop table members;
show tables;
create table members(
    idx int auto_increment primary key,
    uname varchar(20) not null,
    uid varchar(20) not null,
    pwd varchar(20) not null,
    email varchar(50),
    mobile varchar(20),
    apply char(1)
);

show tables;
desc members;
